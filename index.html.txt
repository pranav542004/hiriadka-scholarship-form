  <script>
    document.getElementById('scholarshipForm').addEventListener('submit', async function(e) {
      e.preventDefault();
      const form = e.target;
      const formData = new FormData(form);
      const plainData = {};
      formData.forEach((value, key) => {
        if (typeof value === 'string') {
          plainData[key] = value;
        }
      });

      try {
        const scriptURL = 'https://script.google.com/macros/s/AKfycbxPNZ5ePQWKDXpaCLTqRYek7sM05e2maT7E7NbVArIV8zBtEIqPmg5SVwfTNL0sOYPN/exec';
        const response = await fetch(scriptURL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(plainData)
        });

        if (!response.ok) throw new Error(`HTTP error! Status: ${response.status}`);

        console.log("Saved to sheet");
        localStorage.setItem("scholarshipFormData", JSON.stringify(plainData));
        window.location.href = "success.html";
      } catch (error) {
        console.error("Error saving to sheet", error);
        alert("There was an error saving your data. Please try again later or contact the admin.");
      }
    });
  </script>
